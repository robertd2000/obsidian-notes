## 1
```go

package main

import "fmt"

func a() {
    x := []int{}
    x = append(x, 0) // [0] len = 1, cap = 1
    x = append(x, 1) // [0, 1] len = 2, cap = 2
    x = append(x, 2)// [0, 1, 2]  len = 3, cap = 4
    y := append(x, 3) // [0, 1, 2, 3] len = 4, cap = 4
    z := append(x, 4) // [0, 1, 2, 4] len = 4, cap = 4
    fmt.Println(x, y, z)
    // x = [0 1 2], y = [0 1 2 4], z = [0 1 2 4]
    // т.к. у слайсов y и z один внутренний массив
}

func main() {
    a()
}

```

## 2 

```go

package main

import "fmt"

func main() {
	m := map[string]int{"a": 1, "b": 2, "c": 3}
	
	for a, b := range m {
		fmt.Println(a, b)
	}
}

// a 1
// b 2
// c 3

// порядок случайный (негарантированный)

```

```go

package main

import "fmt"

func main() {
	s := "text"
	fmt.Println(s[0]) // t - первый байт - в виде числа в десятичном виде
	s[0] = "R" // нельзя мутировать строку, будет ошибка компиляции
	fmt.Println(s)
}
```
