## 1
```go

package main

import "fmt"

func main() {
    s := "höllo"
    for i := range s {
        fmt.Printf("position %d: %c\n", i, s[i])
    }
    fmt.Printf("len=%d\n", len(s))
}

// position 0: h
// position 1: Ã
// position 3: l
// position 4: l
// position 5: o
// len=6

// for итерируется по байтам, а не по символам и т.к. ö занимает 2 байта, то выведется его первый байт. Поэтому длина 6, а не 5
```

Вот так все правильно выведется

```go

package main

import "fmt"

func main() {
	s := "höllo"
	for i, v := range s {
		fmt.Printf("position %d: %c\n", i, v)
	}
	fmt.Printf("len=%d\n", len([]rune(s)))
}

// position 0: h
// position 1: ö
// position 3: l
// position 4: l
// position 5: o
// len=5
```
