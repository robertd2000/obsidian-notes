  const res = []

  const m = a2.length

  let i = 0

    res.push(a2[i])

function memo(fn) {
    const hash = new Map()
    
    return function(...args) {
        const key = argsToKey(args)  
        if (hash.has(key)) {
            
            return hash.get(key)
        }

        hash.set(key, res)
        return res
    }
}
